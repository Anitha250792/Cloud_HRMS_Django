class EmployeeAttendanceView( TemplateView ): template_name = ( "employee/attendance/" "my_attendance.html" ) def get_context_data( self, **kwargs ): context = super().get_context_data( **kwargs ) employee = Employee.objects.get( user=self.request.user ) records = Attendance.objects.filter( employee=employee ) working = records.count() present = records.filter( check_in__isnull=False ).count() late = records.filter( is_late=True ).count() lop = records.filter( is_half_day=True ).count() context.update({ "working_days": working, "present_days": present, "late_days": late, "lop_days": lop, "records": records }) return context
